#set page=Line Renderer [Component] #set version=v11 #set imageBig=images/GifCcLineRenderer.gif #set description=This component let you visualize a Curve with Unity's standard LineRenderer Component in Play Mode. If you are updating the curve and visualizing it with this component, you may want to set related 'Math' updateMode parameter to AabbVisible, so the 'Math' and 'Line Renderer' update their data only if it's really needed (e.g. they are visible by any camera).
